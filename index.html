chi = np.zeros((N-1,N))
def Chi2Analysis(i):
    for l in range(N-1):
        for j in range(N):
            addi = []
            for k in range(0,37,1):
                Om = 1-((3/(2*delta[l]))*Ol[j])
                H = (H0[i]*np.sqrt((Om*(1/(1+r[k]))**(-2*delta[l])+
                                    (3/(2*delta[l]))*Ol[j])))
                diff = H - H_data[k]
                d2 = (diff**2/sd[k]**2)
                addi.append(d2)
            c = np.sum(addi)
            chi[l,j] = c
    return chi

def confidence_plot_label(x, y, z,xl,yl,t,fn):
    plt.style.use('classic')
    plt.style.use('seaborn-white')
    plt.figure(figsize=(6,6))
    bf = min([min(r) for r in z])
    index = np.where(z==np.min(z))
    b1 = plt.contourf(x,y,z,levels=[bf,bf+2.3,bf+6.17,bf+11.8],cmap='Greys_r')
    b2 = plt.contour(x,y,z,levels=[bf,bf+2.3,bf+6.17,bf+11.8],
                     colors='k',linestyles='solid',linewidths=1)
    proxy = [plt.Rectangle((0,0),1,1,fc = pc.get_facecolor()[0]) 
             for pc in b1.collections]
    label = [r"$1\sigma$", r"$2\sigma$", r"$3\sigma$"]
    plt.legend(proxy,label)
    plt.plot(x[index[1][0]], y[index[0][0]], marker='*', 
             markersize=15, color="k",mfc='w')
    plt.xlim(min(x)-0.005,max(x)+0.005)
    plt.ylim(min(y)-0.005,max(y)+0.005)
    plt.minorticks_on()
    plt.tick_params(which='major',direction='in',length=7,labeltop=False,
                    labelleft=True,labelright=False,width=2,labelsize=20)
    plt.tick_params(which='minor',direction='in',length=4,width=2)
    plt.xlabel(xl,fontsize=25,fontweight='bold')
    plt.ylabel(yl,fontsize=25,fontweight='bold')
    plt.title(t,fontsize=20,fontweight='bold',loc='right',y=1.0,alpha=0.5
              ,color='k')
    plt.savefig(fn,bbox_inches = 'tight')

def marginal_PDF(x,pdf,xl,yl,t,fn):
    plt.style.use('classic')
    plt.style.use('seaborn-white')
    mean = sum(x *pdf) / sum(pdf)
    sigma = np.sqrt(sum(pdf*(x - mean)**2)/sum(pdf))
    y = norm.pdf(x, mean, sigma)
    y2 = y/sum(y)
    plt.plot(x, y2,'r-',label="$\mu$ = %.4f" %mean+ "\n$\sigma$ = %.4f" %sigma)
    plt.scatter(x,pdf,label="%d" %len(r)+" Data Set")
    plt.legend()
    plt.xlim(min(x),max(x))
    plt.ylim(min(y2),max(y2)+0.01)
    plt.minorticks_on()
    plt.tick_params(which='major',direction='in',length=7,labeltop=False,
                    labelleft=True,labelright=False,width=2,labelsize=20)
    plt.tick_params(which='minor',direction='in',length=4,width=2)
    plt.xlabel(xl,fontsize=20,fontweight='bold')
    plt.ylabel(yl,fontsize=20,fontweight='bold')
    plt.title(t+"\n"+xl+"= %.3f" %mean+" $\pm$ %.3f"%sigma,fontsize=15,
              fontweight='bold',loc='right',y=1.0,alpha=0.5,color='k')
    plt.savefig(fn,bbox_inches = 'tight')
